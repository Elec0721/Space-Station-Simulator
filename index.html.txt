function maxNPCs() {
  return 3 + G.modules.habitat * 5 + G.modules.hull * 2;
}
function dockSlots() { return G.modules.docking; }
function energyRate() {
  const orb = ORBITS[G.orbitIndex];
  return orb.energyBase + G.modules.collector * orb.energyPerCol + G.modules.reactor * 5;
}
function cashRate() {
  let r = 0;
  for (const npc of G.npcs) r += npc.cashPerSec;
  return r;
}
function defenseRating() { return Math.min(90, G.modules.defense * 15); }
function upgradeCost(upg, count) {
  const c = Math.ceil(upg.baseCostCash * Math.pow(upg.costMultCash, count));
  const e = Math.ceil(upg.baseCostEnergy * Math.pow(upg.costMultEnergy, count));
  return { cash: upg.baseCostCash > 0 ? c : 0, energy: upg.baseCostEnergy > 0 ? e : 0 };
}

const canvas = document.getElementById('mainCanvas');
const ctx = canvas.getContext('2d');
let stars = [];
let particles = [];
let npcShips = [];
let orbitAngle = 0;
let stationPulse = 0;

function initCanvas() {
  resize();
  generateStars();
  window.addEventListener('resize', () => { resize(); generateStars(); });
  requestAnimationFrame(renderLoop);
}

function resize() {
  const wrap = document.getElementById('canvas-wrap');
  canvas.width = wrap.clientWidth;
  canvas.height = wrap.clientHeight;
}

function generateStars() {
  stars = [];
  const count = Math.floor(canvas.width * canvas.height / 2500);
  for (let i = 0; i < count; i++) {
    stars.push({
      x: Math.random() * canvas.width,
      y: Math.random() * canvas.height,
      r: Math.random() * 1.5 + 0.2,
      alpha: Math.random() * 0.8 + 0.2,
      twinkle: Math.random() * Math.PI * 2,
      speed: Math.random() * 0.02 + 0.005,
    });
  }
}

function renderLoop(ts) {
  requestAnimationFrame(renderLoop);
  const dt = 1/60;
  orbitAngle += 0.004;
  stationPulse += 0.05;

  ctx.clearRect(0, 0, canvas.width, canvas.height);

  // Background gradient
  const bgGrad = ctx.createRadialGradient(canvas.width/2, canvas.height/2, 0, canvas.width/2, canvas.height/2, canvas.width*0.7);
  bgGrad.addColorStop(0, '#0a1020');
  bgGrad.addColorStop(1, '#030508');
  ctx.fillStyle = bgGrad;
  ctx.fillRect(0, 0, canvas.width, canvas.height);

  drawStars(dt);
  drawOrbitalBody();
  drawStation();
  drawNPCShips(dt);
  drawParticles(dt);
}

function drawStars(dt) {
  for (const s of stars) {
    s.twinkle += s.speed;
    const alpha = s.alpha * (0.6 + 0.4 * Math.sin(s.twinkle));
    ctx.beginPath();
    ctx.arc(s.x, s.y, s.r, 0, Math.PI * 2);
    ctx.fillStyle = `rgba(200, 220, 255, ${alpha})`;
    ctx.fill();
  }
}

function drawOrbitalBody() {
  const orb = ORBITS[G.orbitIndex];
  const cx = canvas.width / 2;
  const cy = canvas.height / 2;
  const dist = Math.min(canvas.width, canvas.height) * 0.36;
  const bx = cx + Math.cos(orbitAngle) * dist;
  const by = cy + Math.sin(orbitAngle) * dist * 0.4;
  const radius = getOrbitalBodyRadius();

  // Draw orbit path
  ctx.beginPath();
  ctx.ellipse(cx, cy, dist, dist * 0.4, 0, 0, Math.PI * 2);
  ctx.strokeStyle = 'rgba(100,160,255,0.08)';
  ctx.lineWidth = 1;
  ctx.stroke();

  if (G.orbitIndex === 0) drawEarth(bx, by, radius);
  else if (G.orbitIndex === 1) drawMars(bx, by, radius);
  else if (G.orbitIndex === 2) drawJupiter(bx, by, radius);
  else if (G.orbitIndex === 3) drawSaturn(bx, by, radius);
  else if (G.orbitIndex === 4) drawSun(bx, by, radius);
  else if (G.orbitIndex === 5) drawStar(bx, by, radius, '#f48fb1');
  else if (G.orbitIndex === 6) drawGalaxy(bx, by, radius);
  else if (G.orbitIndex === 7) drawBlackHole(bx, by, radius);
}

function getOrbitalBodyRadius() {
  const sizes = [30, 22, 38, 35, 50, 28, 42, 25];
  return sizes[G.orbitIndex];
}

function drawEarth(x, y, r) {
  const g = ctx.createRadialGradient(x-r*0.3, y-r*0.3, r*0.1, x, y, r);
  g.addColorStop(0, '#5ab4e8');
  g.addColorStop(0.5, '#2970a0');
  g.addColorStop(1, '#1a4060');
  ctx.beginPath(); ctx.arc(x, y, r, 0, Math.PI*2);
  ctx.fillStyle = g; ctx.fill();
  // continents
  ctx.fillStyle = '#3a7a3a';
  ctx.beginPath(); ctx.ellipse(x-6, y-4, 8, 6, -0.5, 0, Math.PI*2); ctx.fill();
  ctx.beginPath(); ctx.ellipse(x+5, y+3, 6, 5, 0.3, 0, Math.PI*2); ctx.fill();
  // atmosphere glow
  const ag = ctx.createRadialGradient(x, y, r*0.9, x, y, r*1.3);
  ag.addColorStop(0, 'rgba(100,180,255,0.2)');
  ag.addColorStop(1, 'rgba(100,180,255,0)');
  ctx.beginPath(); ctx.arc(x, y, r*1.3, 0, Math.PI*2);
  ctx.fillStyle = ag; ctx.fill();
}

function drawMars(x, y, r) {

  ctx.beginPath(); ctx.ellipse(x+3, y-5, 5, 4, 0.4, 0, Math.PI*2); ctx.fill();
}

function drawSun(x, y, r) {
  // Outer glow
  for (let i = 3; i >= 0; i--) {
    const ag = ctx.createRadialGradient(x, y, r, x, y, r*(1.5+i*0.5));
    ag.addColorStop(0, `rgba(255,200,50,${0.15-i*0.03})`);
    ag.addColorStop(1, 'rgba(255,200,50,0)');
    ctx.beginPath(); ctx.arc(x, y, r*(1.5+i*0.5), 0, Math.PI*2);
    ctx.fillStyle = ag; ctx.fill();
  }
  // Corona rays
  ctx.save();
  ctx.translate(x, y);
  for (let i = 0; i < 12; i++) {
    ctx.rotate(Math.PI*2/12 + orbitAngle*0.3);
    ctx.beginPath(); ctx.moveTo(r*0.9, 0); ctx.lineTo(r*1.6, 0);
    ctx.strokeStyle = 'rgba(255,220,80,0.4)'; ctx.lineWidth = 2; ctx.stroke();
  }
  ctx.restore();
  const g = ctx.createRadialGradient(x-r*0.3, y-r*0.3, r*0.1, x, y, r);
  g.addColorStop(0, '#fffaaa'); g.addColorStop(0.5,'#ffcc00'); g.addColorStop(1, '#ff8800');
  ctx.beginPath(); ctx.arc(x, y, r, 0, Math.PI*2);
  ctx.fillStyle = g; ctx.fill();
}

function drawStar(x, y, r, color) {
  const ag = ctx.createRadialGradient(x, y, 0, x, y, r*2);
  ag.addColorStop(0, color+'88'); ag.addColorStop(1, color+'00');
  ctx.beginPath(); ctx.arc(x, y, r*2, 0, Math.PI*2); ctx.fillStyle = ag; ctx.fill();
  const g = ctx.createRadialGradient(x-r*0.3, y-r*0.3, 1, x, y, r);
  g.addColorStop(0, '#fff'); g.addColorStop(0.4, color); g.addColorStop(1, '#8050a0');
  ctx.beginPath(); ctx.arc(x, y, r, 0, Math.PI*2); ctx.fillStyle = g; ctx.fill();
  // companion star
  const cx2 = x + r*2.5 + Math.cos(orbitAngle*3)*r*0.5;
  const cy2 = y + Math.sin(orbitAngle*3)*r*0.3;
  ctx.beginPath(); ctx.arc(cx2, cy2, r*0.5, 0, Math.PI*2);
  ctx.fillStyle = '#ffe0f0'; ctx.fill();
}

function drawGalaxy(x, y, r) {
  ctx.save(); ctx.translate(x, y); ctx.rotate(orbitAngle*0.1);
  for (let arm = 0; arm < 3; arm++) {
    ctx.save(); ctx.rotate(arm * Math.PI*2/3);
    for (let i = 0; i < 20; i++) {
      const t = i / 20;
      const angle2 = t * Math.PI * 2;
      const dist2 = t * r * 1.4;
      const bx = Math.cos(angle2) * dist2;
      const by = Math.sin(angle2) * dist2 * 0.5;
      ctx.beginPath(); ctx.arc(bx, by, (1-t)*3+0.5, 0, Math.PI*2);
      ctx.fillStyle = `rgba(200,150,255,${0.8-t*0.6})`; ctx.fill();
    }
    ctx.restore();
  }
  ctx.beginPath(); ctx.arc(0, 0, r*0.3, 0, Math.PI*2);
  const g = ctx.createRadialGradient(0,0,0,0,0,r*0.3);
  g.addColorStop(0,'#fff'); g.addColorStop(1,'#cc88ff');
  ctx.fillStyle = g; ctx.fill();
  ctx.restore();
}

function drawBlackHole(x, y, r) {
  // Accretion disk
  for (let i = 0; i < 3; i++) {
    const diskRad = r*(2.5-i*0.4);
    const ag = ctx.createRadialGradient(x, y, r*1.2+i*4, x, y, diskRad);
    ag.addColorStop(0, `rgba(180,60,255,${0.4-i*0.1})`);

    ctx.fillStyle = ag; ctx.fill();
  }
  // Lensing ring
  ctx.beginPath(); ctx.arc(x, y, r*1.5, 0, Math.PI*2);
  ctx.strokeStyle = 'rgba(150,80,255,0.6)'; ctx.lineWidth = 3; ctx.stroke();
  // Event horizon - absolute black
  ctx.beginPath(); ctx.arc(x, y, r, 0, Math.PI*2);
  ctx.fillStyle = '#000'; ctx.fill();
  // Photon sphere glow
  const photon = ctx.createRadialGradient(x, y, r, x, y, r*1.15);
  photon.addColorStop(0, 'rgba(180,60,255,0.8)');
  photon.addColorStop(1, 'rgba(180,60,255,0)');

}

function drawStation() {
  const cx = canvas.width / 2;
  const cy = canvas.height / 2;
  const m = G.modules;
  const scale = 1 + m.hull * 0.12 + m.habitat * 0.04 + m.reactor * 0.03;
  const pulse = Math.sin(stationPulse) * 0.5 + 0.5;

  ctx.save(); ctx.translate(cx, cy); ctx.scale(scale, scale);

  // Core glow
  const cg = ctx.createRadialGradient(0, 0, 5, 0, 0, 35 + pulse*5);
  cg.addColorStop(0, 'rgba(0,229,255,0.3)');
  cg.addColorStop(1, 'rgba(0,229,255,0)');
  ctx.beginPath(); ctx.arc(0, 0, 35+pulse*5, 0, Math.PI*2);
  ctx.fillStyle = cg; ctx.fill();

  // Draw solar panel arms (one per collector, max 8)
  const cols = Math.min(m.collector, 8);
  for (let i = 0; i < cols; i++) {
    const angle = (i / Math.max(cols,1)) * Math.PI * 2;
    const armLen = 30 + i * 3;
    ctx.save(); ctx.rotate(angle);
    ctx.strokeStyle = '#00b8d4'; ctx.lineWidth = 2;
    ctx.beginPath(); ctx.moveTo(0, 0); ctx.lineTo(armLen, 0); ctx.stroke();
    // Panel
    ctx.fillStyle = '#0d47a1'; ctx.strokeStyle = '#40c4ff'; ctx.lineWidth = 1;
    ctx.fillRect(armLen, -8, 18, 16); ctx.strokeRect(armLen, -8, 18, 16);
    ctx.fillStyle = '#1565c0';
    ctx.fillRect(armLen+3, -6, 5, 12); ctx.fillRect(armLen+10, -6, 5, 12);
    ctx.restore();
  }

  // Docking arms (one per bay, max 6)
  const docks = Math.min(m.docking, 6);
  for (let i = 0; i < docks; i++) {
    const angle = (i / Math.max(docks,1)) * Math.PI * 2 + Math.PI/docks;
    ctx.save(); ctx.rotate(angle);
    ctx.strokeStyle = '#607d8b'; ctx.lineWidth = 3;
    ctx.beginPath(); ctx.moveTo(15, 0); ctx.lineTo(40, 0); ctx.stroke();
    // Platform
    ctx.fillStyle = '#1a2a3a'; ctx.strokeStyle = '#546e7a'; ctx.lineWidth = 1;
    ctx.beginPath(); ctx.rect(38, -7, 14, 14); ctx.fill(); ctx.stroke();
    // Dock light
    ctx.beginPath(); ctx.arc(45, -7, 2.5, 0, Math.PI*2);
    ctx.fillStyle = `rgba(0,255,150,${0.5 + 0.5*Math.sin(stationPulse + i)})`; ctx.fill();
    ctx.restore();
  }

  // Habitat rings
  const habs = Math.min(m.habitat, 5);
  for (let i = 0; i < habs; i++) {
    const rr = 22 + i * 10;
    ctx.beginPath();
    ctx.arc(0, 0, rr, 0, Math.PI * 2);
    ctx.strokeStyle = `rgba(100,160,255,${0.4-i*0.06})`; ctx.lineWidth = 4+i; ctx.stroke();
    // Ring segments
    for (let j = 0; j < 8; j++) {
      const a = (j/8)*Math.PI*2;
      ctx.beginPath(); ctx.arc(0,0,rr,a,a+0.25);
      ctx.strokeStyle = `rgba(150,200,255,0.6)`; ctx.lineWidth = 6+i; ctx.stroke();
    }
  }

  // Defense turrets
  const turrets = Math.min(m.defense, 6);
  for (let i = 0; i < turrets; i++) {
    const angle = (i / Math.max(turrets,1)) * Math.PI*2 + Math.PI/4;
    const td = 28;
    const tx = Math.cos(angle) * td;
    const ty = Math.sin(angle) * td;
    ctx.save(); ctx.translate(tx, ty); ctx.rotate(angle + Math.PI/2);
    ctx.fillStyle = '#37474f'; ctx.strokeStyle = '#ff5252'; ctx.lineWidth = 1;
    ctx.fillRect(-4, -4, 8, 10); ctx.strokeRect(-4, -4, 8, 10);
    ctx.fillStyle = '#ff5252';
    ctx.fillRect(-1.5, -6, 3, 4);
    ctx.restore();
  }

  // Reactors
  const reacts = Math.min(m.reactor, 6);
  for (let i = 0; i < reacts; i++) {
    const angle = (i / Math.max(reacts,1)) * Math.PI*2 + Math.PI*0.15;
    const rd = 16;
    const rx = Math.cos(angle) * rd;
    const ry = Math.sin(angle) * rd;
    const rg = ctx.createRadialGradient(rx, ry, 2, rx, ry, 8);
    rg.addColorStop(0, 'rgba(100,255,100,0.8)');
    rg.addColorStop(1, 'rgba(100,255,100,0)');
    ctx.beginPath(); ctx.arc(rx, ry, 8, 0, Math.PI*2); ctx.fillStyle = rg; ctx.fill();
    ctx.beginPath(); ctx.arc(rx, ry, 4, 0, Math.PI*2);
    ctx.fillStyle = `rgba(150,255,150,${0.6+0.4*pulse})`; ctx.fill();
  }

  // CORE HUB
  ctx.rotate(stationPulse * 0.1);
  // Outer ring
  ctx.beginPath(); ctx.arc(0, 0, 18, 0, Math.PI*2);
  ctx.strokeStyle = 'rgba(0,229,255,0.7)'; ctx.lineWidth = 2; ctx.stroke();
  // Core body
  ctx.beginPath();
  for (let i = 0; i < 8; i++) {
    const a = (i/8)*Math.PI*2;
    const rr2 = i%2===0 ? 14 : 10;
    if (i===0) ctx.moveTo(Math.cos(a)*rr2, Math.sin(a)*rr2);
    else ctx.lineTo(Math.cos(a)*rr2, Math.sin(a)*rr2);
  }
  ctx.closePath();
  const coreG = ctx.createRadialGradient(-3,-3,1,0,0,14);
  coreG.addColorStop(0,'#90caf9'); coreG.addColorStop(1,'#0d2040');
  ctx.fillStyle = coreG; ctx.fill();
  ctx.strokeStyle = '#00e5ff'; ctx.lineWidth = 1.5; ctx.stroke();
  // Inner glow
  ctx.beginPath(); ctx.arc(0,0,7,0,Math.PI*2);
  ctx.fillStyle = `rgba(0,229,255,${0.3+0.3*pulse})`; ctx.fill();

  // Shop modules (small squares attached to core)
  const shops = Math.min(m.shop, 4);
  for (let i = 0; i < shops; i++) {
    const angle = (i/4)*Math.PI*2 + Math.PI/8;
    const sx = Math.cos(angle)*12, sy = Math.sin(angle)*12;
    ctx.save(); ctx.translate(sx, sy); ctx.rotate(angle);
    ctx.fillStyle = '#ff8f00'; ctx.strokeStyle = '#ffd54f'; ctx.lineWidth = 1;
    ctx.fillRect(-3,-3,6,6); ctx.strokeRect(-3,-3,6,6);
    ctx.restore();
  }

  ctx.restore(); // main station translate/scale

  // Station label
  ctx.font = `bold 11px 'Orbitron'`;
  ctx.fillStyle = 'rgba(0,229,255,0.7)';
  ctx.textAlign = 'center';
  ctx.fillText('STATION ALPHA', cx, cy - 65 * scale - 10);
}

function drawNPCShips(dt) {
  for (let i = npcShips.length - 1; i >= 0; i--) {
    const s = npcShips[i];
    s.t += dt * s.speed;
    if (s.t > 1) { npcShips.splice(i,1); continue; }
    const x = s.startX + (s.endX - s.startX) * easeInOut(s.t);
    const y = s.startY + (s.endY - s.startY) * easeInOut(s.t);
    ctx.save(); ctx.translate(x, y);
    const angle = Math.atan2(s.endY - s.startY, s.endX - s.startX);
    ctx.rotate(angle);
    // Ship body
    ctx.fillStyle = s.color;
    ctx.beginPath(); ctx.moveTo(8,0); ctx.lineTo(-5,5); ctx.lineTo(-3,0); ctx.lineTo(-5,-5); ctx.closePath();
    ctx.fill();
    // Engine glow
    ctx.beginPath(); ctx.arc(-4, 0, 3, 0, Math.PI*2);
    ctx.fillStyle = `rgba(0,200,255,${0.6*(1-s.t)})`; ctx.fill();
    ctx.restore();
  }
}

function spawnNPCShip(npc) {
  const cx = canvas.width/2, cy = canvas.height/2;
  const edge = Math.random() < 0.5;
  const sx = edge ? (Math.random()<0.5 ? -20 : canvas.width+20) : Math.random()*canvas.width;
  const sy = edge ? Math.random()*canvas.height : (Math.random()<0.5 ? -20 : canvas.height+20);
  npcShips.push({
    startX:sx, startY:sy, endX:cx+Math.random()*40-20, endY:cy+Math.random()*40-20,
    t:0, speed:0.4+Math.random()*0.3,
    color: ['#00e5ff','#69f0ae','#ffd54f','#ea80fc','#ff7043'][Math.floor(Math.random()*5)]
  });
}

function drawParticles(dt) {
  for (let i = particles.length-1; i >= 0; i--) {
    const p = particles[i];
    p.life -= dt;
    if (p.life <= 0) { particles.splice(i,1); continue; }
    const alpha = p.life / p.maxLife;
    p.x += p.vx * dt; p.y += p.vy * dt;
    ctx.beginPath(); ctx.arc(p.x, p.y, p.r * alpha, 0, Math.PI*2);
    ctx.fillStyle = p.color.replace('1)', `${alpha})`); ctx.fill();
  }
}

function spawnParticles(x, y, color, count) {
  for (let i = 0; i < count; i++) {
    const angle = Math.random() * Math.PI * 2;
    const speed = Math.random() * 60 + 20;
    particles.push({
      x, y, vx:Math.cos(angle)*speed, vy:Math.sin(angle)*speed,
      r:3+Math.random()*3, life:0.8+Math.random()*0.5, maxLife:1.3,
      color:`rgba(${color},1)`
    });
  }
}

function easeInOut(t) { return t < 0.5 ? 2*t*t : -1+(4-2*t)*t; }

function gameTick() {
  if (G.gameWon) return;
  G.ticks++;

  // Energy income
  const er = energyRate();
  G.energy = Math.min(G.maxEnergy, G.energy + er);

  // Cash income from NPCs
  for (const npc of G.npcs) {
    G.cash += npc.cashPerSec;
    npc.timeLeft -= 1;
  }

  // Remove NPCs whose time is up
  const leaving = G.npcs.filter(n => n.timeLeft <= 0);
  for (const n of leaving) {
    addLog(`${n.icon} ${n.name} departed.`, 'info');
  }
  G.npcs = G.npcs.filter(n => n.timeLeft > 0);

  // NPC arrivals
  G.nextNpcIn--;
  if (G.nextNpcIn <= 0) {
    if (G.npcs.length < maxNPCs()) {
      spawnNPC();
    }
    const interval = Math.max(5, 20 - G.modules.docking * 2);
    G.nextNpcIn = interval + Math.floor(Math.random() * interval);
  }

  // Random events (check every 30 ticks)
  if (G.ticks % 30 === 0) {
    const chance = 0.12 * ORBITS[G.orbitIndex].eventMult;
    if (Math.random() < chance) triggerRandomEvent();
  }

  // Win condition: reached black hole
  if (G.orbitIndex === ORBITS.length - 1 && G.modules.hull >= 5 && G.modules.collector >= 10) {
    if (!G.gameWon) winGame();
  }

  updateUI();
}

function spawnNPC() {
  const type = NPC_TYPES[Math.floor(Math.random() * NPC_TYPES.length)];
  const bonusCash = G.modules.shop * 10;
  const bonusRate = G.modules.shop * 2;
  const arrival = type.cashOnArrive + bonusCash;
  G.cash += arrival;

  const npc = {
    id: G.npcIdCounter++,
    name: type.name,
    icon: type.icon,
    cashPerSec: type.cashPerSec + bonusRate,
    timeLeft: 30 + Math.floor(Math.random() * 60),
    maxTime: 30 + Math.floor(Math.random() * 60),
  };
  G.npcs.push(npc);
  spawnNPCShip(npc);
  spawnParticles(canvas.width/2, canvas.height/2, '0,229,255', 8);
  addLog(`${npc.icon} ${npc.name} arrived! (+${fmt(arrival)} 💰)`, 'cash');
}

function triggerRandomEvent() {
  const ev = EVENTS[Math.floor(Math.random() * EVENTS.length)];
  const defense = defenseRating() / 100;
  const orbitMult = 1 + G.orbitIndex * 0.3;

  if (ev.type === 'bad') {
    const dmg = Math.floor(ev.baseDmg * orbitMult * (1 - defense));
    if (ev.cashLoss) {
      const lost = Math.min(G.cash, dmg);
      G.cash -= lost;
      showAlert(`${ev.icon} ${ev.name}: -${fmt(lost)} 💰 ${defense>0?`(${Math.round(defense*100)}% blocked)`:''}`, 'bad');
      addLog(`${ev.icon} ${ev.name}: -${fmt(lost)} 💰`, 'warn');
    }
    if (ev.energyLoss) {
      const lost = Math.min(G.energy, dmg);
      G.energy -= lost;
      showAlert(`${ev.icon} ${ev.name}: -${fmt(lost)} ⚡`, 'bad');
      addLog(`${ev.icon} ${ev.name}: -${fmt(lost)} ⚡`, 'warn');
    }
  } else {
    if (ev.cashBonus) {
      const bonus = Math.floor(100 * orbitMult);
      G.cash += bonus;
      showAlert(`${ev.icon} ${ev.name}: +${fmt(bonus)} 💰`, 'good');
      addLog(`${ev.icon} ${ev.name}: +${fmt(bonus)} 💰`, 'good');
    }
    if (ev.energyBonus) {
      const bonus = Math.floor(150 * orbitMult);
      G.energy = Math.min(G.maxEnergy, G.energy + bonus);
      showAlert(`${ev.icon} ${ev.name}: +${fmt(bonus)} ⚡`, 'good');
      addLog(`${ev.icon} ${ev.name}: +${fmt(bonus)} ⚡`, 'good');
    }
  }
}

}

function buildUpgradeList() {
  const list = document.getElementById('upgrade-list');
  list.innerHTML = '';
  for (const upg of UPGRADES) {
    const card = document.createElement('div');
    card.className = 'upgrade-card';
    card.id = 'upg-' + upg.id;
    card.innerHTML = `
      <div class="upg-header">
        <span class="upg-name">${upg.icon} ${upg.name}</span>
        <span class="upg-count" id="cnt-${upg.id}">0</span>
      </div>
      <div class="upg-desc">${upg.desc}</div>
      <div class="upg-cost" id="cost-${upg.id}"></div>
      <div class="upg-effect">${upg.effect}</div>

    `;
    card.addEventListener('click', () => buyUpgrade(upg.id));
    list.appendChild(card);
  }
  updateUpgradeCards();
}

function updateUpgradeCards() {
  for (const upg of UPGRADES) {
    const count = G.modules[upg.id] || 0;
    const card = document.getElementById('upg-' + upg.id);
    if (!card) continue;
    const cnt = document.getElementById('cnt-' + upg.id);
    const costEl = document.getElementById('cost-' + upg.id);
    if (count >= upg.maxCount) {
      cnt.textContent = 'MAX';
      costEl.innerHTML = '<span style="color:var(--cyan2)">MAXED OUT</span>';
      card.classList.add('disabled');
      card.classList.remove('can-afford');
      continue;
    }
    const cost = upgradeCost(upg, count);
    cnt.textContent = count;
    let costHTML = '';
    if (cost.cash > 0) costHTML += `<span class="cost-item cost-cash">💰 ${fmt(cost.cash)}</span>`;
    if (cost.energy > 0) costHTML += `<span class="cost-item cost-energy">⚡ ${fmt(cost.energy)}</span>`;
    costEl.innerHTML = costHTML;
    const canAfford = G.cash >= cost.cash && G.energy >= cost.energy;
    card.classList.toggle('can-afford', canAfford);
    card.classList.toggle('disabled', false);
  }
}

function buyUpgrade(id) {
  if (G.gameWon) return;
  const upg = UPGRADES.find(u => u.id === id);
  if (!upg) return;
  const count = G.modules[id] || 0;
  if (count >= upg.maxCount) return;
  const cost = upgradeCost(upg, count);
  if (G.cash < cost.cash || G.energy < cost.energy) {
    showAlert('⚠ Insufficient resources!', 'bad');
    return;
  }
  G.cash -= cost.cash;
  G.energy -= cost.energy;
  G.modules[id] = count + 1;
  if (upg.id === 'reactor') G.maxEnergy += 500;
  spawnParticles(canvas.width/2, canvas.height/2, '0,229,255', 12);
  addLog(`Upgraded: ${upg.icon} ${upg.name} → Lv.${G.modules[id]}`, 'good');
  updateUpgradeCards();
  updateUI();
}

function buildOrbitButtons() {

function updateOrbitButtons() {
  for (let i = 0; i < ORBITS.length; i++) {
    const btn = document.getElementById('orbit-btn-' + i);
    if (!btn) continue;
    btn.classList.remove('orbit-current','orbit-locked');
    const cost = ORBIT_COSTS[i];
    if (i === G.orbitIndex) {
      btn.classList.add('orbit-current');
    } else if (i < G.orbitIndex || (G.cash >= cost.cash && G.energy >= cost.energy && i === G.orbitIndex + 1)) {
      // accessible (already visited or next affordable)
    } else {
      btn.classList.add('orbit-locked');
    }
  }
}

function confirmMigration(index) {
  if (index <= G.orbitIndex) return;
  if (index !== G.orbitIndex + 1) {
    showAlert('⚠ Must migrate to next orbit first!', 'bad');
    return;
  }
  const orb = ORBITS[index];
  const cost = ORBIT_COSTS[index];
  if (G.cash < cost.cash || G.energy < cost.energy) {
    showAlert('⚠ Insufficient resources for migration!', 'bad');
    return;
  }
  G.pendingModal = { type:'migrate', index };
  document.getElementById('modal-title').textContent = `MIGRATE TO ${orb.name.toUpperCase()}`;
  document.getElementById('modal-body').innerHTML = `
    Move Station Alpha to orbit ${orb.name}.<br>
    <em style="color:var(--text2);font-size:11px;">${orb.desc}</em>
    <div class="modal-cost">
      <div style="color:var(--text2);font-size:11px;margin-bottom:4px;">MIGRATION COST:</div>
      <div class="modal-cost-row">
        <span class="cost-item cost-cash">💰 ${fmt(cost.cash)} Credits</span>
        <span class="cost-item cost-energy">⚡ ${fmt(cost.energy)} Energy</span>
      </div>
    </div>
    <div style="margin-top:10px;font-size:11px;color:var(--green);">
      ↑ Energy rate: ×${orb.energyMult} (base ${orb.energyBase}/s)
    </div>
  `;
  document.getElementById('modal-overlay').classList.add('active');
}

function doMigrate(index) {
  const cost = ORBIT_COSTS[index];
  G.cash -= cost.cash;
  G.energy -= cost.energy;
  G.orbitIndex = index;
  const orb = ORBITS[index];
  // All NPCs leave on migration
  G.npcs = [];
  addLog(`🚀 Migrated to ${orb.name} orbit!`, 'good');
  showAlert(`🚀 Arrived at ${orb.name}! Energy ×${orb.energyMult}`, 'good');
  spawnParticles(canvas.width/2, canvas.height/2, '105,240,174', 30);
  updateOrbitButtons();
  updateUI();
  // Check win after reaching black hole
  if (index === ORBITS.length - 1) {
    addLog('🕳️ You orbit a Black Hole. The end... or the beginning.', 'warn');
  }
}

let alertTimer = null;
function showAlert(msg, type='bad') {
  const el = document.getElementById('event-alert');
  el.textContent = msg;
  el.className = 'show' + (type === 'good' ? ' good' : '');
  if (alertTimer) clearTimeout(alertTimer);
  alertTimer = setTimeout(() => { el.className = ''; }, 3500);

function fmt(n) {
  if (n === undefined || n === null) return '0';
  n = Math.floor(n);
  if (n >= 1e9) return (n/1e9).toFixed(2) + 'B';
  if (n >= 1e6) return (n/1e6).toFixed(2) + 'M';
  if (n >= 1e3) return n.toLocaleString();
  return n.toString();
}

function fmtShort(n) {
  if (n >= 1e6) return (n/1e6).toFixed(1)+'M';
  if (n >= 1e3) return (n/1e3).toFixed(0)+'K';
  return n.toString();
}
